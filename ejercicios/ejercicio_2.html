import pandas as pd

assistents = ["carme","raúl","pau","pep","carles","julia","julia","julia","carme","joan","pep","joan", "carme"]
llista_temp = []

unics = set(assistents)

for a in unics:
    persona = a
    aparicions = assistents.count(a)
    tupla = (persona, aparicions)
    llista_temp.append(tupla)

print(llista_temp)

df = pd.DataFrame(list(llista_temp), columns=["assistents","sessions"])

df["promig"] = sum(df.sessions)/len(df)
df["desviacio"] = df["sessions"]-df["promig"]

print(df)
